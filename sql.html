عرض اسم وراتب كل موظف

SELECT name, salary

سهل

عرض الموظفين في قسم المبيعات فقط.

النتيجة المتوقعة:

جميع الموظفين في قسم المبيعات

SELECT *

4. استخدام BETWEEN و AND

متوسط

عرض الموظفين الذين رواتبهم بين 3000 و 5000.

النتيجة المتوقعة:

الموظفين برواتب في النطاق المحدد

SELECT name, salary 
FROM employees 
WHERE salary BETWEEN 3000 AND 5000;

5. استخدام LIKE للبحث النصي

متوسط

البحث عن الموظفين الذين أسماؤهم تبدأ بـ "محمد".

النتيجة المتوقعة:

جميع الموظفين الذين أسماؤهم تبدأ بـ "محمد"

SELECT name 
FROM employees 
WHERE name LIKE 'محمد%';

6. استخدام ORDER للترتيب

متوسط

عرض الموظفين مرتبين تنازلياً حسب الراتب.

النتيجة المتوقعة:

الموظفين من الأعلى راتباً إلى الأقل

SELECT name, salary 
FROM employees 
ORDER BY salary DESC;

7. استخدام COUNT

سهل

حساب عدد الموظفين في الشركة.

النتيجة المتوقعة:

عدد الموظفين الإجمالي

SELECT COUNT(*) AS total_employees 
FROM employees;

8. استخدام SUM و AVG

سهل

حساب مجموع ومتوسط رواتب الموظفين.

النتيجة المتوقعة:

المجموع والمتوسط للرواتب

SELECT 
    SUM(salary) AS total_salaries,
    AVG(salary) AS average_salary 
FROM employees;

9. استخدام GROUP BY

متوسط

حساب عدد الموظفين في كل قسم.

النتيجة المتوقعة:

عدد الموظفين مصنف حسب القسم

SELECT 
    department,
    COUNT(*) AS employee_count 
FROM employees 
GROUP BY department;

10. استخدام HAVING

متوسط

عرض الأقسام التي بها أكثر من 5 موظفين.

النتيجة المتوقعة:

الأقسام التي لديها 5 موظفين أو أكثر

SELECT 
    department,
    COUNT(*) AS employee_count 
FROM employees 
GROUP BY department 
HAVING COUNT(*) > 5;

11. استخدام INNER JOIN

متوسط

ربط جدول الموظفين بجدول الأقسام.

الجداول:

employees و departments

SELECT 
    e.name, 
    e.salary, 
    d.department_name 
FROM employees e 
INNER JOIN departments d ON e.department_id = d.id;

12. استخدام LEFT JOIN

متوسط

عرض جميع الموظفين حتى لو لم يكونوا في قسم.

النتيجة المتوقعة:

جميع الموظفين مع أسماء أقسامهم (إن وجدت)

SELECT 
    e.name, 
    d.department_name 
FROM employees e 
LEFT JOIN departments d ON e.department_id = d.id;

13. استخدام SELF JOIN

صعب

عرض الموظفين مع مديريهم.

الجدول: employees

يحتوي على manager_id يشير إلى id المدير

SELECT 
    e.name AS employee_name,
    m.name AS manager_name 
FROM employees e 
LEFT JOIN employees m ON e.manager_id = m.id;

14. استخدام Subquery مع IN

متوسط

عرض الموظفين الذين رواتبهم أعلى من المتوسط.

النتيجة المتوقعة:

الموظفين برواتب أعلى من متوسط الرواتب

SELECT name, salary 
FROM employees 
WHERE salary > (
    SELECT AVG(salary) 
    FROM employees
);

15. استخدام Correlated Subquery

صعب

عرض الموظفين الذين رواتبهم أعلى من متوسط قسمهم.

النتيجة المتوقعة:

الموظفين برواتب أعلى من متوسط قسمهم

SELECT name, department, salary 
FROM employees e1 
WHERE salary > (
    SELECT AVG(salary) 
    FROM employees e2 
    WHERE e2.department = e1.department
);

16. استخدام الدوال النصية

سهل

تحويل أسماء الموظفين إلى أحرف كبيرة.

النتيجة المتوقعة:

الأسماء بحروف كبيرة

SELECT 
    UPPER(name) AS employee_name,
    salary 
FROM employees;

17. استخدام الدوال التاريخية

متوسط

حساب عدد سنوات عمل كل موظف.

النتيجة المتوقعة:

اسم الموظف وعدد سنوات الخبرة

SELECT 
    name,
    DATEDIFF(YEAR, hire_date, GETDATE()) AS years_of_service 
FROM employees;

18. إنشاء جدول

سهل

إنشاء جدول جديد للموظفين.

هيكل الجدول:

id, name, email, salary, department_id, hire_date

CREATE TABLE employees (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE,
    salary DECIMAL(10,2),
    department_id INT,
    hire_date DATE,
    FOREIGN KEY (department_id) REFERENCES departments(id)
);

20. تحديث بيانات

متوسط

زيادة رواتب جميع موظفي قسم المبيعات بنسبة 10%.

التأثير:

زيادة رواتب قسم المبيعات

UPDATE employees 
SET salary = salary * 1.10 
WHERE department_id = (
    SELECT id FROM departments 
    WHERE department_name = 'Sales'
);
