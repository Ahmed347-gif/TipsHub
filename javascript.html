4. المضروب

متوسط

حساب مضروب رقم n.

المدخلات والمخرجات:

المدخل: 5

المخرج: 120

function factorial(n) {
    if (n === 0 || n === 1) return 1;
    let result = 1;
    for (let i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}

// مثال على الاستخدام
console.log(factorial(5)); // 120

6. التحقق من العدد الأولي

متوسط

المدخل: 17

المخرج: true

function isPrime(num) {
    if (num <= 1) return false;
    if (num <= 3) return true;
    
    if (num % 2 === 0 || num % 3 === 0) return false;
    
    for (let i = 5; i * i <= num; i += 6) {
        if (num % i === 0 || num % (i + 2) === 0) return false;
    }
    
    return true;
}

// مثال على الاستخدام
console.log(isPrime(17)); // true

9. متسلسلة فيبوناتشي

متوسط

اطبع أول n أعداد في سلسة فيبوناتشي.

المدخلات والمخرجات:

المدخل: 7

المخرج: [0, 1, 1, 2, 3, 5, 8]

function fibonacci(n) {
    const sequence = [0, 1];
    
    for (let i = 2; i < n; i++) {
        sequence[i] = sequence[i - 1] + sequence[i - 2];
    }
    
    return sequence.slice(0, n);
}

// مثال على الاستخدام
console.log(fibonacci(7)); // [0, 1, 1, 2, 3, 5, 8]

10. القاسم المشترك الأكبر

متوسط

احسب القاسم المشترك الأكبر.

المدخلات والمخرجات:

المدخل: 56, 98

المخرج: 14

function gcd(a, b) {
    // Using Euclidean Algorithm
    while (b !== 0) {
        const temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

// مثال على الاستخدام
console.log(gcd(56, 98)); // 14

12. عد الحروف الصوتية

سهل

احسب عدد الحروف الصوتية في نص.

المدخلات والمخرجات:

المدخل: "Hello World"

المخرج: 3

function countVowels(str) {
    const vowels = ['a', 'e', 'i', 'o', 'u'];
    let count = 0;
    
    for (let char of str.toLowerCase()) {
        if (vowels.includes(char)) {
            count++;
        }
    }
    
    return count;
}

// مثال على الاستخدام
console.log(countVowels("Hello World")); // 3

17. ترتيب المصفوفة

صعب

رتب مصفوفة باستخدام Bubble Sort.

المدخلات والمخرجات:

المدخل: [5, 2, 8, 1, 9]

المخرج: [1, 2, 5, 8, 9]

function bubbleSort(arr) {
    const n = arr.length;
    
    for (let i = 0; i < n - 1; i++) {
        for (let j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                // Swap elements
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
            }
        }
    }
    
    return arr;
}

// مثال على الاستخدام
console.log(bubbleSort([5, 2, 8, 1, 9])); // [1, 2, 5, 8, 9]

20. عد تكرار حرف

المدخلات والمخرجات:

المدخل: "programming", "g"

المخرج: 2

function countCharacter(str, char) {
    let count = 0;
    for (let c of str) {
        if (c === char) {
            count++;
        }
    }
    return count;
}

// مثال على الاستخدام
console.log(countCharacter("programming", "g")); // 2
